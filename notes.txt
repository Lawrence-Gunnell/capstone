#on any deb based linux machine you can install postgres with the following command
sudo apt install postgresql 

#you can also run the official postgres docker image https://hub.docker.com/_/postgres
docker run --name some-postgres -e POSTGRES_PASSWORD=mysecretpassword -d postgres

#to access postgres. it will ask for your postgres user password. you'll likely need to exec into the docker container if you go the docker route. 
psql -U postgres -h localhost
